/**********************************************************************
			    数码管静态显示 					   	
**********************************************************************/

#include<pic.h>  //头文件
//__CONFIG(0x3B31);//MPLAB和PICC的连接语
/********************************************************************
                数据类型定义
*********************************************************************/
#define uchar unsigned char 
#define uint unsigned int
/********************************************************************
                数码管段码定义0123456789ABCDEF
*********************************************************************/
uchar Table[16]={0xc0,0Xf9,0xa4,0xb0,0x99,0x92,0x82,0xf8,0x80,0x90,0X08,0x03,0xC6,0xA1,0x06,0x0E};	 //所需的段的位码
/**********************************************************************
				  MS级延时函数程序，参数i 延时时间								   	
**********************************************************************/
void DelayMS(uint i)
{
uint j;
for(;i!=0;i--)
{
for(j=8000;j!=0;j--);
}
}
//////////////////////////////初始化////////////////////////////////////
void Init_IO(void)//初始化I/O
{
//设置B口为输出
TRISB=0x00;
PORTB=0xFF;
//设置D口为输出
TRISD=0x00;
PORTD=0xFF;
}
/**********************************************************************
				           主函数						   	
**********************************************************************/
void main(void)
{
Init_IO();  //初始化I/O

while(1)
{
PORTB=Table[0];//显示0；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[1];//显示1；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[2];//显示2；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[3];//显示3；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[4];//显示4；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[5];//显示5；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[6];//显示6；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[7];//显示7；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[8];//显示8；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[9];//显示9；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[10];//显示A；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[11];//显示b；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[12];//显示C；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[13];//显示d；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[14];//显示E；
PORTD=0x0F;
DelayMS(20);
PORTB=Table[15];//显示F；
PORTD=0x0F;
DelayMS(20);
}
}

/**********************************************************************
		                   结束						   	
**********************************************************************/
