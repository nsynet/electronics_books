#include<pic.h>  //头文件
//__CONFIG(0x3B31);//MPLAB和PICC的连接语
/**********************************************************************
				  函数数据类型说明							   	
**********************************************************************/
#define uchar unsigned char 
#define uint unsigned int
/**********************************************************************
				  MS级延时函数程序，参数i 延时时间								   	
**********************************************************************/
void DelayMS(uint i)
{
uint j;
for(;i!=0;i--)
{
for(j=8000;j!=0;j--);
}
}
/*****************************************************
                    I/O初始化
*****************************************************/
void INTO()
{ 
	TRISD=0x00;//定义单片机D口为输出
	PORTD=0xff;
}

/*********************************************************************
				   主函数						   	
**********************************************************************/
void main()     //主函数
{
	INTO();
	while(1)
	{

	PORTD=0xff;
	DelayMS(10);
	PORTD=0x00;
	DelayMS(10);
	}	
}
/**********************************************************************
				                 结束					   	
**********************************************************************/
