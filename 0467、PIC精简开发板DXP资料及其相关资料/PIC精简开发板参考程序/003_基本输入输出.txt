#include<pic.h>  //头文件
//__CONFIG(0x3B31);//MPLAB和PICC的连接语

/*****************************************************
                    I/O初始化
*****************************************************/
void INTO()
{ 
	TRISC=0Xff;//定义单片机A口为输入
	PORTC=0X00;
	TRISD=0x00;//定义单片机D口为输出
	PORTD=0xff;
}

/*********************************************************************
				   主函数						   	
**********************************************************************/
void main()     //主函数
{
	INTO();
	while(1)
	{
	
		if(RC0==0){PORTD=0x0f;}
		if(RC1==0){PORTD=0xf0;}

	}	
}
/**********************************************************************
				                 结束					   	
**********************************************************************/
