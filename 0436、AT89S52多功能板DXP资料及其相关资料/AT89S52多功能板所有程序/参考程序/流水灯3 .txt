    #include<reg52.h>

#include <intrins.h>


void delayms(unsigned char ms) // 延时子程序
{      
 unsigned char i;
 while(ms--)
 {
  for(i = 0; i < 120; i++);
 }
}

main()
{
 unsigned char LED;
 LED = 0xff;
 P0 = LED;
 
 while(1)
 {
  delayms(1000);   
  LED = LED<<1;  //循环右移1位，点亮下一个LED
 
  if(LED==0x00)
  {
    LED=0xff;
	}
   P0 = LED;
 }
}
